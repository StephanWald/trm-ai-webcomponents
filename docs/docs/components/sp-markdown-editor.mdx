# sp-markdown-editor

Full-featured markdown editor web component with source, WYSIWYG, and split view modes. Includes a formatting toolbar, syntax highlighting (via Prism.js), voice dictation, file import/export, undo/redo history, and auto-save support.

## Peer Dependencies

Full WYSIWYG and preview mode functionality requires the following peer dependencies to be loaded before the component:

```html
<!-- Markdown parsing -->
<script src="https://cdn.jsdelivr.net/npm/marked/marked.min.js"></script>
<!-- HTML sanitization -->
<script src="https://cdn.jsdelivr.net/npm/dompurify/dist/purify.min.js"></script>
<!-- Syntax highlighting -->
<script src="https://cdn.jsdelivr.net/npm/prismjs/prism.min.js"></script>
```

Without these, the editor works fully in **source mode** but WYSIWYG/preview rendering will be limited.

## Installation

```bash
npm install @skillspilot/webcomponents marked dompurify prismjs
```

Or load from CDN:

```html
<script src="https://cdn.jsdelivr.net/npm/marked/marked.min.js"></script>
<script src="https://cdn.jsdelivr.net/npm/dompurify/dist/purify.min.js"></script>
<script src="https://cdn.jsdelivr.net/npm/prismjs/prism.min.js"></script>
<script type="module" src="https://cdn.jsdelivr.net/npm/@skillspilot/webcomponents/dist/skillspilot/skillspilot.esm.js"></script>
```

### Event Handling

Listen to `contentChange`, `save`, `modeChange`, `importFile`, and `exportFile` events.

<LiveExample height="420px" code={`
<script src="https://cdn.jsdelivr.net/npm/marked/marked.min.js"><\/script>
<script src="https://cdn.jsdelivr.net/npm/dompurify/dist/purify.min.js"><\/script>
<sp-markdown-editor id="editor4" auto-save="true" auto-save-delay="3000"></sp-markdown-editor>
<div id="evtlog" style="margin-top:8px;padding:8px;background:#f5f5f5;border-radius:4px;font-family:monospace;font-size:0.75rem;min-height:60px;"></div>
<script>
  const editor = document.getElementById('editor4');
  const log = document.getElementById('evtlog');

  function addLog(msg) {
    const p = document.createElement('p');
    p.style.margin = '2px 0';
    p.textContent = new Date().toLocaleTimeString() + ' - ' + msg;
    log.prepend(p);
    if (log.children.length > 6) log.lastChild.remove();
  }

  editor.value = '# Event Demo\n\nEdit this text and watch the events below.';

  editor.addEventListener('contentChange', e => addLog('contentChange: ' + (e.detail ? e.detail.length : 0) + ' chars'));
  editor.addEventListener('save', e => addLog('save: ' + (e.detail && e.detail.content ? e.detail.content.length : 0) + ' chars'));
  editor.addEventListener('modeChange', e => addLog('modeChange: ' + e.detail));
<\/script>
`} />
